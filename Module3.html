Question 1 - Bar Plot Counting Characters¶

Code creating bar plot¶

import matplotlib.pyplot as plt
num_list = [] # Count of Character
char_list = [] # Character

for x,y in sorted(list(letter_freq.items())): # Sort to alphabetize the list
    char_list.append(x)
    num_list.append(y)

fig = plt.figure(figsize = (10,5)) 
ax = fig.add_subplot()
ax.bar(char_list, num_list)
plt.xlabel('Character', fontsize = 14)
plt.ylabel('Frequency', fontsize = 14)
plt.title('Frequency of Characters', fontsize = 18)
plt.savefig('char.png')
plt.show()

fig, ax = plt.subplots(figsize=(10,5))

lns1 = ax.plot(sizes,insertion_sort_times,color="red", marker="o", label = 'Insertion Sort Times')

ax2 = ax.twinx()
lns2 = ax2.plot(sizes,merge_sort_times, color="blue", marker="o",label = 'Merge Sort Times')

lns = lns1+lns2
labs = [l.get_label() for l in lns]
ax.legend(lns, labs, loc=0)

ax.set_xlabel("Sizes")
ax.set_ylabel("Insertion Sort Times", color="red", fontsize=14)
